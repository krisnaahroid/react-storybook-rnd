## USAGE

Import your `Input component` in your file

```jsx
import { InputLabel } from "../your_path/components";

```

```jsx
import React, { useState } from 'react'

const Example = () => {
    const [user, setUser] = useState('');

    const handlerUserID = e => {
        setUser(e.target.value)
    }

    return (
        <InputLabel 
        title="User ID" 
        type="text" 
        value={user} 
        width="100%" 
        bottom="20px"
        onChange={ handlerUserID }
        />
    )

}
        
```
